Events
{
  static
    Events *eventListener  // a pointer to the object which receives the static pollEvents dispatch
    SUCCESS pollEvents() // loops through polled events and calls eventListener's handleEvent
    
  members
    handleEvent(SDL_Event *event) - virtual  // parses the event struct and calls the appropriate interface virtual
                                             // overrides of this function (which may choose to dispatch the event to other objects) can call the Events::handleEvent as a default of sorts  
    ...event interface... - virtual and empty  // a series of functions which represent the various events the user may want to define behavior for.
}
